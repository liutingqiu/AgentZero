## 需求

在右侧面板已有的"步骤时间轴"分区中，增加看板任务的创建和状态管理交互。

### 当前状态

右侧面板已有 `renderRightTimeline()` 函数，从 `/api/kanban` GET 获取任务列表并渲染为时间轴。目前是只读的。

### 需要新增的功能

#### 功能 1：在时间轴顶部添加"新建任务"输入框

在时间轴分区标题下方，添加一行：

```
┌──────────────────────────────┐
│ 步骤时间轴                    │
│ ┌───────────────────┐ [添加] │
│ │ 输入任务标题...     │        │
│ └───────────────────┘        │
│ ● 任务1                 ✓    │
│ ● 任务2                      │
└──────────────────────────────┘
```

点击"添加"按钮 → `POST /api/kanban` → 成功后刷新时间轴 + `showToast`。
按 Enter 键在输入框中也可以触发添加。

#### 功能 2：每个任务项增加"完成"切换按钮

在每个任务项的右侧，增加一个小按钮（圆形勾选图标）：

- 状态为 `done` → 显示实心勾 ✓（点击可取消完成）
- 状态为 `pending` / `running` → 显示空心勾 ○（点击标记完成）

点击 → `PUT /api/kanban/{id}` 更新状态 → 成功后刷新时间轴 + `showToast`。

#### 功能 3：每个任务项增加"删除"按钮

每个任务项右侧增加一个 ✕ 删除按钮。

点击 → 调用 `confirmDialog` 确认 → 确认后 `DELETE /api/kanban/{id}` → 成功后刷新时间轴 + `showToast`。

### JS 部分（`app.js` 文件末尾追加）

**新增函数**：

```javascript
// ===== 看板 CRUD 交互 =====
async function kanbanCreateTask(title) {
  try {
    const resp = await fetch('/api/kanban', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ title: title })
    });
    const data = await resp.json();
    if (data.ok) {
      showToast('任务已创建', 'success');
      renderRightTimeline();
    } else {
      showToast(data.error?.msg || '创建失败', 'error');
    }
  } catch (e) {
    showToast('网络错误: ' + e.message, 'error');
  }
}

async function kanbanToggleTask(id, currentStatus) {
  const newStatus = currentStatus === 'done' ? 'pending' : 'done';
  try {
    const resp = await fetch('/api/kanban/' + id, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ status: newStatus })
    });
    const data = await resp.json();
    if (data.ok) {
      showToast(newStatus === 'done' ? '任务已完成' : '任务已恢复', 'success');
      renderRightTimeline();
    } else {
      showToast(data.error?.msg || '更新失败', 'error');
    }
  } catch (e) {
    showToast('网络错误: ' + e.message, 'error');
  }
}

async function kanbanDeleteTask(id) {
  const confirmed = await confirmDialog('确定要删除此任务吗？');
  if (!confirmed) return;
  try {
    const resp = await fetch('/api/kanban/' + id, {
      method: 'DELETE'
    });
    const data = await resp.json();
    if (data.ok) {
      showToast('任务已删除', 'info');
      renderRightTimeline();
    } else {
      showToast(data.error?.msg || '删除失败', 'error');
    }
  } catch (e) {
    showToast('网络错误: ' + e.message, 'error');
  }
}
```

### 示例：修改后的 renderRightTimeline 伪代码

```javascript
function renderRightTimeline() {
  const section = document.getElementById('rightTimeline');
  if (!section) return;

  // 从 /api/kanban 获取数据
  fetch('/api/kanban')
    .then(r => r.json())
    .then(data => {
      const tasks = data.tasks || data || [];
      let html = '<div class="right-panel-section-title">步骤时间轴</div>';

      // 输入行
      html += '<div class="kanban-input-row">'
        + '<input class="kanban-input" id="kanbanInput" placeholder="新任务标题..." onkeydown="if(event.key===\'Enter\'){var v=this.value.trim();if(v){kanbanCreateTask(v);this.value=\'\';}}">'
        + '<button class="kanban-add-btn" onclick="var inp=document.getElementById(\'kanbanInput\');var v=inp.value.trim();if(v){kanbanCreateTask(v);inp.value=\'\';}">添加</button>'
        + '</div>';

      // 任务列表
      if (tasks.length === 0) {
        html += '<div style="font-size:11px;color:var(--muted);padding:8px 0;">暂无任务</div>';
      } else {
        tasks.forEach(t => {
          const isDone = t.status === 'done';
          html += '<div class="kanban-item">'
            + '<button class="kanban-toggle-btn' + (isDone ? ' done' : '') + '" onclick="kanbanToggleTask(' + t.id + ',\'' + t.status + '\')">'
            + (isDone ? '✓' : '') + '</button>'
            + '<span class="kanban-item-title' + (isDone ? ' done' : '') + '">' + escapeHtml(t.title || '') + '</span>'
            + '<button class="kanban-delete-btn" onclick="kanbanDeleteTask(' + t.id + ')">✕</button>'
            + '</div>';
        });
      }

      section.innerHTML = html;
    })
    .catch(() => {
      section.innerHTML = '<div class="right-panel-section-title">步骤时间轴</div>'
        + '<div style="font-size:11px;color:#fb2c36;padding:8px 0;">加载失败</div>';
    });
}
```
