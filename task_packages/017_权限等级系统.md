### 三级权限定义

| 等级 | 文件读取 | 文件写入 | Shell/命令执行 | 说明 |

**1. 在输入区下方添加等级选择器**

找到输入区的 HTML 构建位置（约第 100-121 行的 `input-area`），在 `.input-hint` 之前或之后插入等级选择器。

三个按钮样式类似单选按钮组，当前选中的高亮：

```
[ Plan ] [ Auto ] [ YOLO ]
```

```javascript
// 在文件顶部状态管理中新增：
let currentPermissionLevel = localStorage.getItem('zero_permission') || 'plan';

// 在 send() 函数附近或合适位置新增函数：
function renderPermissionSelector() {
  const container = document.getElementById('permissionSelector');
  if (!container) return;
  const levels = ['plan', 'auto', 'yolo'];
  container.innerHTML = levels.map(l => {
    const active = l === currentPermissionLevel;
    return '<button class="perm-btn perm-' + l + (active ? ' active' : '') + '" onclick="setPermission(\'' + l + '\')">' + l.toUpperCase() + '</button>';
  }).join('');
}

function setPermission(level) {
  currentPermissionLevel = level;
  localStorage.setItem('zero_permission', level);
  renderPermissionSelector();
  // yolo 模式下输入框边框变红警告
  const inp = document.getElementById('inp');
  if (inp) {
    inp.closest('.input-box').classList.toggle('yolo-warning', level === 'yolo');
  }
}

// 在页面初始化时调用：
renderPermissionSelector();
```

```javascript
// 新增函数：
function showApprovalDialog(command, requestId) {
  const overlay = document.createElement('div');
  overlay.className = 'approval-overlay';
  overlay.innerHTML =
    '<div class="approval-box">'
    + '<div class="approval-title">⚠️ 命令执行审批</div>'
    + '<div class="approval-cmd">' + escapeHtml(command) + '</div>'
    + '<div class="approval-actions">'
    + '<button class="approval-btn approval-deny" id="approvalDeny">拒绝</button>'
    + '<button class="approval-btn approval-allow" id="approvalAllow">允许</button>'
    + '</div></div>';
  document.body.appendChild(overlay);

  document.getElementById('approvalDeny').onclick = function() {
    overlay.remove();
    fetch('/api/approval/' + requestId, { method: 'POST', body: JSON.stringify({approved: false}) });
  };
  document.getElementById('approvalAllow').onclick = function() {
    overlay.remove();
    fetch('/api/approval/' + requestId, { method: 'POST', body: JSON.stringify({approved: true}) });
  };
}
```

#### tools.py

为需要权限门控的工具函数添加权限检查：

```python
# 在文件顶部或工具函数中定义
PERMISSION_LEVEL = 'plan'  # 由外部设置

def set_permission_level(level):
    global PERMISSION_LEVEL
    PERMISSION_LEVEL = level

def _check_file_read(path):
    """plan 模式下只允许读取项目目录下的文件"""
    import os
    if PERMISSION_LEVEL == 'yolo':
        return True
    if PERMISSION_LEVEL == 'plan':
        # 只允许读取项目目录和 data/ 目录
        allowed_dirs = [os.path.abspath('.'), os.path.abspath('data')]
        abs_path = os.path.abspath(path)
        for d in allowed_dirs:
            if abs_path.startswith(d):
                return True
        return False
    return True  # auto 模式允许读写

def _check_file_write(path):
    """写入权限：plan 禁止，auto/yolo 允许"""
    if PERMISSION_LEVEL == 'plan':
        return False
    return True

def _check_command(cmd):
    """命令执行权限：plan 禁止，auto 需审批，yolo 允许"""
    if PERMISSION_LEVEL == 'plan':
        return False, None
    if PERMISSION_LEVEL == 'auto':
        return False, 'approval_required'  # 需要用户审批
    return True, None  # yolo 允许
```

#### sandbox.py

新增权限检查函数（可选，如果已有类似函数则复用）：

```python
def check_permission(level: str, action: str, target: str = '') -> tuple[bool, str]:
    """检查权限等级是否允许执行某操作
    
    Args:
        level: plan / auto / yolo
        action: read / write / execute
        target: 操作目标路径或命令
    
    Returns:
        (allowed, reason) — allowed 为 False 时 reason 说明原因
    """
    if level == 'yolo':
        return True, 'ok'
    if level == 'plan':
        if action == 'execute':
            return False, 'plan 模式下禁止执行命令'
        if action == 'write':
            return False, 'plan 模式下禁止写入文件'
        # read 操作由调用方做路径白名单检查
        return True, 'ok'
    # auto 模式
    if action == 'execute':
        return False, 'approval_required'  # 需要用户审批
    return True, 'ok'
```

## 验收标准

| # | 验收项 | 操作 |
|:-:|-------|------|
| 1 | plan 模式默认选中 | 新用户打开页面，默认选中 plan |
| 2 | plan 模式只读 | 让零执行写文件操作 → 返回权限不足提示 |
| 3 | auto 模式 shell 审批 | 让零执行一个 shell 命令 → 前端弹出审批窗口 → 点允许后执行 |
| 4 | yolo 模式红色警告 | 切换到 yolo → 输入框边框变红 → 零可以执行任何操作 |
| 5 | 权限持久化 | 刷新页面 → 权限保持上一次选择 |
| 6 | 消息携带权限 | 检查 SSE URL 中包含 `&perm=plan` 参数 |
