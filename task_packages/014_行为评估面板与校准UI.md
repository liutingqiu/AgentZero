## 需求

### 后端：新增 2 个行为评估 API 端点

#### 端点 1：`GET /api/behavior/control` — 当前控制强度状态

返回当前全量或按 task_type 筛选的控制强度值、温度策略等。

```python
# 路由注册（在 make_app 中）
app.router.add_get('/api/behavior/control', handle_behavior_control)

# handler
async def handle_behavior_control(request):
    """获取当前行为控制强度配置"""
    task_type = request.query.get('type', '').strip()

    try:
        from behavior_canon import compute_control_strength, get_agent_residual
        TASK_TYPES = ['code', 'planning', 'analysis', 'chat', 'reasoning', 'text', 'search']

        if task_type:
            strength = compute_control_strength(task_type)
            # 获取对应 agent 残差
            agent_id_map = {'code': 'reasonix', 'planning': 'zero', 'analysis': 'zero',
                           'chat': 'agnes_text', 'reasoning': 'zero', 'text': 'agnes_text', 'search': 'tavily'}
            agent_id = agent_id_map.get(task_type, 'zero')
            residual = get_agent_residual(agent_id)
            return web.json_response({
                'ok': True,
                'task_type': task_type,
                'control_strength': strength,
                'temperature': 0.2 + (1.0 - strength) * 0.6,  # 温度策略简化公式
                'agent_residual': residual
            })
        else:
            # 返回所有类型的控制强度
            result = {}
            for tt in TASK_TYPES:
                strength = compute_control_strength(tt)
                agent_id = agent_id_map.get(tt, 'zero')
                result[tt] = {
                    'control_strength': strength,
                    'temperature': round(0.2 + (1.0 - strength) * 0.6, 4),
                    'agent': agent_id
                }
            return web.json_response({'ok': True, 'controls': result})

    except Exception as e:
        return _error_response(ERR_INTERNAL, str(e), status=500)
```

**查询参数**：
- `type` — 任务类型（可选，不传则返回全部）

**成功响应**（全部）：
```json
{
  "ok": true,
  "controls": {
    "code": {"control_strength": 0.85, "temperature": 0.29, "agent": "reasonix"},
    "chat": {"control_strength": 0.20, "temperature": 0.68, "agent": "agnes_text"},
    ...
  }
}
```

#### 端点 2：`GET /api/behavior/report` — 校准报告

返回行为校准历史摘要或当前校准状态。

```python
# 路由注册
app.router.add_get('/api/behavior/report', handle_behavior_report)

# handler
async def handle_behavior_report(request):
    """获取行为校准报告"""
    try:
        from behavior_canon import _memory, _calibrator

        # 从校准器获取报告数据
        report = {
            'total_calibrations': 0,
            'recent_actions': [],
            'current_profile': {}
        }

        # 尝试从 _memory 和 _calibrator 获取数据
        if _calibrator is not None and hasattr(_calibrator, 'get_report'):
            report.update(_calibrator.get_report())
        elif _memory is not None and hasattr(_memory, 'get_recent_actions'):
            report['recent_actions'] = _memory.get_recent_actions(10)

        return web.json_response({'ok': True, 'report': report})
    except Exception as e:
        return _error_response(ERR_INTERNAL, str(e), status=500)
```

### 前端：在右侧面板增加"行为评估"分区

#### JS 部分（`app.js` 文件末尾追加）

```javascript
// ===== 行为评估面板 =====
function renderRightBehavior() {
  const section = document.getElementById('rightBehavior');
  if (!section) return;

  // 请求控制强度数据
  fetch('/api/behavior/control')
    .then(r => r.json())
    .then(data => {
      if (!data.ok) {
        section.innerHTML = '<div class="right-panel-section-title">行为评估</div><div style="font-size:11px;color:var(--muted);padding:8px 0;">暂无数据</div>';
        return;
      }

      let html = '<div class="right-panel-section-title">行为评估</div>';

      if (data.controls) {
        // 显示各任务类型的控制强度条形图
        const types = Object.keys(data.controls);
        types.forEach(t => {
          const c = data.controls[t];
          const barWidth = Math.round(c.control_strength * 100);
          const tempColor = c.temperature > 0.6 ? '#fbbf24' : (c.temperature > 0.4 ? '#3b82f6' : '#4ade80');
          html += '<div class="behavior-row">'
            + '<div class="behavior-label">' + t + '</div>'
            + '<div class="behavior-bar-bg">'
            + '<div class="behavior-bar" style="width:' + barWidth + '%;background:' + tempColor + ';"></div>'
            + '</div>'
            + '<div class="behavior-value">' + barWidth + '%</div>'
            + '</div>';
        });
      } else if (data.control_strength !== undefined) {
        // 单个类型
        const sw = Math.round(data.control_strength * 100);
        html += '<div class="behavior-row">'
          + '<div class="behavior-label">' + data.task_type + '</div>'
          + '<div class="behavior-bar-bg">'
          + '<div class="behavior-bar" style="width:' + sw + '%;"></div>'
          + '</div>'
          + '<div class="behavior-value">' + sw + '%</div>'
          + '</div>';
        html += '<div style="font-size:11px;color:var(--muted);margin-top:4px;">温度: ' + data.temperature + '</div>';
      }

      section.innerHTML = html;
    })
    .catch(function () {
      section.innerHTML = '<div class="right-panel-section-title">行为评估</div><div style="font-size:11px;color:#fb2c36;padding:8px 0;">加载失败</div>';
    });
}
```
