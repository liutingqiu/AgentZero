## 需求

### 后端：新增 3 个记忆 API 端点

在 `server.py` 中新增以下路由和 handler。

#### 端点 1：`GET /api/memory` — 获取记忆列表

```python
# 路由注册
app.router.add_get('/api/memory', handle_memory_list)

# handler
async def handle_memory_list(request):
    """获取持久化记忆列表"""
    query = request.query.get('q', '').strip()
    limit = request.query.get('limit', '20').strip()
    try:
        limit = max(1, min(100, int(limit)))
    except ValueError:
        limit = 20

    try:
        if query:
            memories = memory_manager.search_tasks(query)
        else:
            memories = memory_manager.list_persistent_memories()
        # 确保 memories 是可迭代的
        if memories is None:
            memories = []
        # 截取 limit 条
        memories = list(memories)[:limit]
        return web.json_response({'ok': True, 'memories': memories})
    except Exception as e:
        return _error_response(ERR_INTERNAL, str(e), status=500)
```

**查询参数**：
- `q` — 搜索关键词（可选）
- `limit` — 返回条数上限，默认 20，最大 100

**成功响应**：
```json
{"ok": true, "memories": [{"id": 1, "title": "...", "summary": "...", "created_at": "..."}]}
```

#### 端点 2：`POST /api/memory` — 保存新记忆

```python
# 路由注册
app.router.add_post('/api/memory', handle_memory_create)

# handler
async def handle_memory_create(request):
    """保存一条新记忆"""
    try:
        body = await request.json()
    except Exception:
        return _error_response(ERR_BAD_REQUEST, '请求格式错误')

    title = body.get('title', '').strip()
    content = body.get('content', '').strip()
    if not title or not content:
        return _error_response(ERR_BAD_REQUEST, '标题和内容不能为空')

    try:
        memory = memory_manager.save_persistent_memory(title, content)
        return web.json_response({'ok': True, 'memory': memory})
    except Exception as e:
        return _error_response(ERR_INTERNAL, str(e), status=500)
```

**请求体**：
```json
{"title": "关键发现", "content": "用户反馈中提到了性能瓶颈..."}
```

**成功响应**：
```json
{"ok": true, "memory": {"id": 2, "title": "关键发现", "content": "..."}}
```

#### 端点 3：`DELETE /api/memory/{id}` — 删除记忆

```python
# 路由注册
app.router.add_delete('/api/memory/{id}', handle_memory_delete)

# handler
async def handle_memory_delete(request):
    """删除一条记忆"""
    mem_id = request.match_info.get('id')
    if not mem_id or not mem_id.isdigit():
        return _error_response(ERR_BAD_REQUEST, '无效的记忆 ID')

    try:
        result = memory_manager.delete_memory(int(mem_id))
        if not result:
            return _error_response(ERR_NOT_FOUND, f'记忆 {mem_id} 不存在')
        return web.json_response({'ok': True})
    except Exception as e:
        return _error_response(ERR_INTERNAL, str(e), status=500)
```

**memory_manager 实例获取**：
检查 `server.py` 中是否已经创建了 `memory_manager` 实例（可能在 `make_app()` 函数中或模块级别）。如果没有，在 `make_app()` 中导入并创建：

```python
from cognition.memory_manager import MemoryManager
memory_manager = MemoryManager()
```

注意：不要重复创建，如果已有实例就直接引用。

### 前端：在右侧面板增加"记忆"分区

#### JS 部分（`app.js` 文件末尾追加）

在现有的 `renderRightPanel` 函数中，增加第 4 个分区的渲染调用（在时间轴之后）：

```javascript
function renderRightMemory() {
  const section = document.getElementById('rightMemory');
  if (!section) return;

  fetch('/api/memory?limit=10')
    .then(r => r.json())
    .then(data => {
      if (!data.ok || !data.memories) {
        section.innerHTML = '<div class="right-panel-section-title">记忆</div><div style="font-size:11px;color:var(--muted);padding:8px 0;">暂无记忆</div>';
        return;
      }
      let html = '<div class="right-panel-section-title">记忆</div>';
      html += '<div class="memory-search-row">'
        + '<input class="memory-search-input" id="memorySearchInput" placeholder="搜索记忆..." onkeydown="if(event.key===\'Enter\'){memorySearch(this.value);}">'
        + '</div>';

      if (data.memories.length === 0) {
        html += '<div style="font-size:11px;color:var(--muted);padding:8px 0;">暂无记忆</div>';
      } else {
        data.memories.forEach(m => {
          html += '<div class="memory-item">'
            + '<div class="memory-item-title">' + escapeHtml(m.title || '') + '</div>'
            + '<div class="memory-item-summary">' + escapeHtml((m.summary || m.content || '').slice(0, 60)) + '</div>'
            + '<button class="memory-delete-btn" onclick="memoryDelete(' + m.id + ')">✕</button>'
            + '</div>';
        });
      }
      section.innerHTML = html;
    })
    .catch(() => {
      section.innerHTML = '<div class="right-panel-section-title">记忆</div><div style="font-size:11px;color:#fb2c36;padding:8px 0;">加载失败</div>';
    });
}

// 记忆搜索函数
function memorySearch(query) {
  const section = document.getElementById('rightMemory');
  if (!section) return;
  fetch('/api/memory?q=' + encodeURIComponent(query) + '&limit=10')
    .then(r => r.json())
    .then(data => {
      if (!data.ok) return;
      let html = '<div class="right-panel-section-title">记忆 · 搜索</div><div style="font-size:11px;color:var(--muted);padding:4px 0;">"' + query + '" 的搜索结果：</div>';
      if (data.memories.length === 0) {
        html += '<div style="font-size:11px;color:var(--muted);padding:8px 0;">无匹配结果</div>';
      } else {
        data.memories.forEach(m => {
          html += '<div class="memory-item">'
            + '<div class="memory-item-title">' + escapeHtml(m.title || '') + '</div>'
            + '<div class="memory-item-summary">' + escapeHtml((m.summary || m.content || '').slice(0, 60)) + '</div>'
            + '<button class="memory-delete-btn" onclick="memoryDelete(' + m.id + ')">✕</button>'
            + '</div>';
        });
      }
      section.innerHTML = html;
      // 添加返回按钮
      const backBtn = document.createElement('button');
      backBtn.className = 'memory-back-btn';
      backBtn.textContent = '← 返回全部';
      backBtn.onclick = renderRightMemory;
      section.appendChild(backBtn);
    });
}

function memoryDelete(id) {
  confirmDialog('确定要删除这条记忆吗？').then(confirmed => {
    if (!confirmed) return;
    fetch('/api/memory/' + id, { method: 'DELETE' })
      .then(r => r.json())
      .then(data => {
        if (data.ok) {
          showToast('记忆已删除', 'info');
          renderRightMemory();
        } else {
          showToast(data.error?.msg || '删除失败', 'error');
        }
      })
      .catch(e => showToast('网络错误', 'error'));
  });
}
```
